\begin{problem}[5 bodů]
Najděte všechna $n$ taková, že $C_n$ je izomorfní svému doplňku $\overline{C_n}$  a dokažte, že žádná další neexistují. Kružnici $C_n$ definujeme pouze pro $n \geq 3$.
\end{problem}

\begin{problem}[5 bodů]
Rozhodněte, pro která $k \geq 3$ existují přirozená čísla $m$, $n$ taková, že úplný bipartitní graf $K_{m,n}$ obsahuje cyklus $C_k$ jako podgraf.

\noindent
\textit{Hint: Pokud si nevíte rady, tak nejprve prozkoumejte tvrzení pro $k=3$ a $k=4$.}

\end{problem}
